3. PHP + MySQL

<?php
$conn = mysqli_connect("localhost","root","","medica");
if(!$conn) die("Błąd połączenia");

$wynik = mysqli_query($conn,"SELECT * FROM abonamenty");
while($row = mysqli_fetch_assoc($wynik)){
    echo "<p>".$row['nazwa']." - ".$row['cena']." zł</p>";
}
?>
